#include <Servo.h>

Servo servo1; 
Servo servo2;  

void setup() {
  servo1.attach(8);  
  servo2.attach(9);   
}

void loop() {
  for (int pos = 0; pos <= 180; pos++) {
    servo1.write(pos);          
    servo2.write(180 - pos); 
    delay(15);                  
  }

  for (int pos = 180; pos >= 0; pos--) {
    servo1.write(pos);         
    servo2.write(180 - pos);    
    delay(15);
  }
}
